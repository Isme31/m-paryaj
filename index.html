function robotLogic() { const robotSym = "O"; const playerSym = mySymbol; // "X" let bestScore = -Infinity; let move = -1; // 1. Chèche tout kaz ki vid let emptyCells = []; for (let i = 0; i < 100; i++) { if (boardData[i] === "") emptyCells.push(i); } if (emptyCells.length === 0) return; // 2. Analize chak kaz pou wè kote ki pi enpòtan for (let i of emptyCells) { let score = 0; // --- DEFANS: Bloke jwè a (Trè Enpòtan) --- let dangerLevel = evaluateMove(i, playerSym); if (dangerLevel >= 1000) score += 5000; // Si jwè a pral genyen, BLOKE L KOUNYE A! else score += dangerLevel * 2; // Bay defans la plis pwa // --- ATAK: Chèche genyen (Offensif) --- let attackLevel = evaluateMove(i, robotSym); if (attackLevel >= 1000) score += 10000; // Si robo a ka genyen, l ap jwe l an premye else score += attackLevel; if (score > bestScore) { bestScore = score; move = i; } } // 3. Poze pyon an if (move !== -1) { boardData[move] = robotSym; renderBoard(); if (checkWin5(move, robotSym)) { id('status').innerText = "Robo a BAT OU! 🤖🔥"; myTurn = false; } else { myTurn = true; id('status').innerText = "Tou pa w!"; } } } // Fonksyon entèlijan pou evalye danje oswa chans sou yon kaz function evaluateMove(index, s) { const size = 10; const r = Math.floor(index / size); const c = index % size; const dirs = [[0, 1], [1, 0], [1, 1], [1, -1]]; let maxPotential = 0; for (let [dr, dc] of dirs) { let count = 1; // Gade devan for (let i = 1; i < 5; i++) { let nr = r + dr * i, nc = c + dc * i; if (nr >= 0 && nr < 10 && nc >= 0 && nc < 10 && boardData[nr * 10 + nc] === s) count++; else break; } // Gade dèyè for (let i = 1; i < 5; i++) { let nr = r - dr * i, nc = c - dc * i; if (nr >= 0 && nr < 10 && nc >= 0 && nc < 10 && boardData[nr * 10 + nc] === s) count++; else break; } // Sistèm pwen agresif if (count >= 5) maxPotential += 10000; // Viktwa else if (count === 4) maxPotential += 1000; // 4 aliye (Danje lanmò) else if (count === 3) maxPotential += 100; // 3 aliye (Dwe bloke) else if (count === 2) maxPotential += 10; } return maxPotential; }
